# DB_SCHEMA.txt

Компактная схема БД для MVP Telegram-бота на Aiogram 3.0.

Идея:
- берем твою схему USER_PROFILE + ROADMAP за основу;
- не делаем отдельный большой каталог материалов, потому что LLM имеет доступ к интернету и сама ищет бесплатные русскоязычные ресурсы;
- добавляем только те таблицы, которые реально нужны для сценария хакатона: шапка маршрута, элементы маршрута, фидбек/коррекция, лог LLM, топ пользователей.

Рекомендуемая СУБД: PostgreSQL 15+.

CREATE EXTENSION IF NOT EXISTS pgcrypto;


1. USER_PROFILE

Главная таблица пользователя.
Здесь хранится Telegram-пользователь, образовательный портрет, состояние диалога и краткая геймификация.

CREATE TABLE USER_PROFILE (
    User_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    Telegram_id BIGINT NOT NULL UNIQUE,
    Username TEXT,
    First_name TEXT,
    Last_name TEXT,

    Goal_text TEXT,                         -- Что пользователь хочет освоить
    Direction TEXT,                         -- Сфера: IT, дизайн, маркетинг, кулинария и т.д.
    Specific_track TEXT,                    -- Уточнение: Backend, UX/UI, SMM и т.д.
    Target_role TEXT,                       -- Желаемая роль: Frontend-разработчик, дизайнер и т.д.
    Goal_reason TEXT,                       -- Зачем пользователь учится

    Current_level TEXT CHECK (
        Current_level IN ('beginner', 'basic', 'professional')
    ),                                      -- beginner / basic / professional

    Time_per_week_label TEXT,               -- До 3 часов / 3-7 часов / Более 7 часов
    Time_per_week_value INT,                -- Числовая оценка часов в неделю: 2 / 5 / 9

    Preferred_formats TEXT[] DEFAULT '{}',  -- video / practice / article
    Wishes TEXT,                            -- Сырой текст пожеланий пользователя
    Preference_json JSONB NOT NULL DEFAULT '{
        "collected": false,
        "hard_rules": [],
        "soft_rules": [],
        "blocked_authors": [],
        "blocked_channels": [],
        "blocked_sources": [],
        "preferred_sources": [],
        "format_weights": {
            "video": 1.0,
            "article": 1.0,
            "practice": 1.0
        },
        "max_video_minutes": null,
        "explanation_style": [],
        "pace_preference": "normal"
    }'::jsonb,                              -- Структурированные ограничения и предпочтения
    Notification_settings_json JSONB NOT NULL DEFAULT '{
        "push_enabled": true,
        "motivation_style": "duolingo_aggressive",
        "quiet_hours": {
            "enabled": true,
            "start": "22:00",
            "end": "09:00"
        },
        "max_pushes_per_day": 3,
        "deadline_pushes": true,
        "streak_pushes": true,
        "xp_pushes": true,
        "allow_roast": true
    }'::jsonb,                              -- Настройки мотивационных пушей

    Global_xp INT NOT NULL DEFAULT 0,
    Streak_days INT NOT NULL DEFAULT 0,
    Streak_multiplier NUMERIC(4, 2) NOT NULL DEFAULT 1.0,
    Last_activity DATE,

    Dialog_state TEXT NOT NULL DEFAULT 'start',
    Profile_json JSONB NOT NULL DEFAULT '{}'::jsonb,

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    Updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_user_profile_telegram_id ON USER_PROFILE (Telegram_id);
CREATE INDEX idx_user_profile_dialog_state ON USER_PROFILE (Dialog_state);
CREATE INDEX idx_user_profile_target_role ON USER_PROFILE (Target_role);
CREATE INDEX idx_user_profile_preference_json ON USER_PROFILE USING GIN (Preference_json);
CREATE INDEX idx_user_profile_notification_settings ON USER_PROFILE USING GIN (Notification_settings_json);


Почему эта таблица нужна:
- бот должен помнить пользователя между сообщениями;
- нужно хранить ответы на вопросы Prompt 1;
- Wishes хранит исходный текст пожеланий, а Preference_json хранит правила для LLM;
- Notification_settings_json хранит стиль и лимиты мотивационных пушей;
- XP/streak нужны для мотивации и интеграции с "Прогрессорами";
- Profile_json позволяет сохранить полный JSON от LLM без расширения схемы при каждом новом поле.

Как использовать Preference_json:
- hard_rules: строгие запреты, например не показывать видео или конкретного автора;
- soft_rules: мягкие предпочтения, например больше практики или короче материалы;
- blocked_authors / blocked_channels / blocked_sources: черный список материалов;
- preferred_sources: любимые источники пользователя;
- format_weights: веса video/article/practice для ранжирования;
- max_video_minutes: лимит длины видео;
- explanation_style: желаемый стиль объяснения;
- pace_preference: normal / short_steps / intensive / relaxed.

Как использовать Notification_settings_json:
- motivation_style = "duolingo_aggressive" означает дерзкие и настойчивые пуши;
- пуши не должны оскорблять пользователя, давить на личные темы или нарушать quiet_hours;
- max_pushes_per_day ограничивает частоту, чтобы мотивация не превращалась в спам.


2. ROADMAP

Шапка маршрута.
Один пользователь может создать несколько маршрутов или пересоздать старый, поэтому маршрут лучше вынести отдельно от профиля.

CREATE TABLE ROADMAP (
    Roadmap_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Profile_id UUID NOT NULL REFERENCES USER_PROFILE(User_id) ON DELETE CASCADE,

    Title TEXT NOT NULL,                    -- Название маршрута
    Direction TEXT,                         -- Направление
    Target_role TEXT,                       -- Роль или цель
    Level TEXT,                             -- Уровень, под который построен маршрут

    Estimated_duration_weeks INT,           -- Примерная длительность
    Hours_per_week_label TEXT,              -- Темп пользователя
    Route_logic TEXT,                       -- Почему шаги идут в такой последовательности

    Status TEXT NOT NULL DEFAULT 'active' CHECK (
        Status IN ('draft', 'active', 'paused', 'completed', 'replaced', 'archived')
    ),

    Version INT NOT NULL DEFAULT 1,         -- Увеличивается при сильной коррекции маршрута
    Roadmap_json JSONB NOT NULL DEFAULT '{}'::jsonb,

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    Updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_roadmap_profile_status ON ROADMAP (Profile_id, Status);
CREATE INDEX idx_roadmap_target_role ON ROADMAP (Target_role);


Почему эта таблица нужна:
- если хранить только строки материалов, будет негде сохранить название, общую логику и статус маршрута;
- можно пересоздавать маршрут, не теряя старый;
- Roadmap_json хранит полный ответ LLM для вывода карты/mini app.


3. ROADMAP_ITEM

Конкретные шаги маршрута: навык, тема, материал, практика или итоговый проект.
Это доработанная версия твоей таблицы ROADMAP.

CREATE TABLE ROADMAP_ITEM (
    Item_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Roadmap_id UUID NOT NULL REFERENCES ROADMAP(Roadmap_id) ON DELETE CASCADE,
    Profile_id UUID NOT NULL REFERENCES USER_PROFILE(User_id) ON DELETE CASCADE,

    Step_order INT NOT NULL,                -- Общий порядок в маршруте
    Skill_name TEXT NOT NULL,               -- Навык: HTML/CSS, композиция, SQL и т.д.
    Topic_name TEXT,                        -- Тема внутри навыка

    Name TEXT NOT NULL,                     -- Название материала или задания
    Description TEXT,                       -- Что изучить на этом шаге
    Resources TEXT,                         -- URL или список URL через перенос строки

    Source_type TEXT NOT NULL CHECK (
        Source_type IN ('video', 'text', 'practice', 'course', 'lecture', 'article', 'collection', 'project')
    ),
    Source_name TEXT,                       -- Stepik, YouTube, Хабр, VK Видео и т.д.

    Is_free BOOLEAN NOT NULL DEFAULT true,
    Language TEXT NOT NULL DEFAULT 'ru',
    Difficulty TEXT CHECK (
        Difficulty IN ('beginner', 'basic', 'intermediate', 'advanced')
    ),

    Duration_minutes INT,                   -- Для видео/лекций
    Estimated_hours NUMERIC(5, 2),          -- Оценка времени на шаг

    Why_this_material TEXT,                 -- Почему подходит пользователю
    Skill_result TEXT,                      -- Что пользователь сможет после шага
    Career_value TEXT,                      -- К каким задачам/ролям приближает
    Practice_task TEXT,                     -- Практика после материала
    Self_check_questions JSONB DEFAULT '[]'::jsonb,
    Completion_check_type TEXT NOT NULL DEFAULT 'self_check' CHECK (
        Completion_check_type IN ('self_check', 'quiz', 'practice', 'note', 'project', 'manual')
    ),                                      -- Как подтверждается прохождение
    Completion_check_json JSONB NOT NULL DEFAULT '{}'::jsonb,
    Min_seconds_before_complete INT NOT NULL DEFAULT 0,

    Issued_at TIMESTAMPTZ DEFAULT now(),
    Recommended_deadline_at TIMESTAMPTZ,
    Deadline_at TIMESTAMPTZ,

    Xp INT NOT NULL DEFAULT 0,
    Pending_xp INT NOT NULL DEFAULT 0,       -- XP ожидает проверки
    Verified_xp INT NOT NULL DEFAULT 0,      -- XP подтверждено тестом/практикой
    Xp_policy_json JSONB NOT NULL DEFAULT '{}'::jsonb,
    Fraud_score NUMERIC(5, 2) NOT NULL DEFAULT 0,
    Streak_multiplier NUMERIC(4, 2) NOT NULL DEFAULT 1.0,

    Status TEXT NOT NULL DEFAULT 'not_started' CHECK (
        Status IN ('not_started', 'in_progress', 'pending_check', 'completed', 'completed_late', 'expired', 'skipped', 'replaced')
    ),

    User_note TEXT,
    Completed_at TIMESTAMPTZ,
    Item_json JSONB NOT NULL DEFAULT '{}'::jsonb,

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    Updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

    UNIQUE (Roadmap_id, Step_order)
);

CREATE INDEX idx_roadmap_item_roadmap_order ON ROADMAP_ITEM (Roadmap_id, Step_order);
CREATE INDEX idx_roadmap_item_profile_status ON ROADMAP_ITEM (Profile_id, Status);
CREATE INDEX idx_roadmap_item_deadline ON ROADMAP_ITEM (Deadline_at);


Почему эта таблица нужна:
- маршрут состоит из многих шагов, а не из одной записи;
- пользователь должен отмечать каждый материал как пройденный;
- Completion_check_json хранит мини-тест, вопрос самопроверки или критерии практики;
- Min_seconds_before_complete защищает от мгновенного закрытия материалов;
- Pending_xp и Verified_xp разделяют "попытку получить XP" и реально подтвержденный XP;
- дедлайн, XP и статус должны быть у конкретного материала/задания;
- можно заменить один материал, не пересоздавая весь маршрут.


4. ROADMAP_FEEDBACK

Фидбек пользователя по материалам и маршруту.

CREATE TABLE ROADMAP_FEEDBACK (
    Feedback_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Profile_id UUID NOT NULL REFERENCES USER_PROFILE(User_id) ON DELETE CASCADE,
    Roadmap_id UUID NOT NULL REFERENCES ROADMAP(Roadmap_id) ON DELETE CASCADE,
    Item_id UUID REFERENCES ROADMAP_ITEM(Item_id) ON DELETE SET NULL,

    Feedback_type TEXT NOT NULL CHECK (
        Feedback_type IN ('useful', 'not_suitable', 'too_hard', 'too_easy', 'already_completed', 'change_request')
    ),
    Feedback_text TEXT,

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_roadmap_feedback_profile ON ROADMAP_FEEDBACK (Profile_id);
CREATE INDEX idx_roadmap_feedback_item ON ROADMAP_FEEDBACK (Item_id);
CREATE INDEX idx_roadmap_feedback_type ON ROADMAP_FEEDBACK (Feedback_type);


Почему эта таблица нужна:
- по требованиям хакатона пользователь должен давать обратную связь;
- один материал может получить несколько реакций;
- по этой таблице LLM сможет понять, что заменить: слишком сложно, уже пройдено, не подходит и т.д.


5. LLM_RUN

Лог вызовов LLM.
Для MVP можно не показывать пользователю, но для отладки и презентации это полезно.

CREATE TABLE LLM_RUN (
    Run_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Profile_id UUID REFERENCES USER_PROFILE(User_id) ON DELETE SET NULL,
    Roadmap_id UUID REFERENCES ROADMAP(Roadmap_id) ON DELETE SET NULL,

    Prompt_name TEXT NOT NULL CHECK (
        Prompt_name IN ('profile_analysis', 'roadmap_generation', 'roadmap_correction')
    ),

    Input_json JSONB NOT NULL,
    Output_json JSONB,

    Status TEXT NOT NULL DEFAULT 'success' CHECK (
        Status IN ('success', 'failed')
    ),
    Error_text TEXT,

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_llm_run_profile ON LLM_RUN (Profile_id);
CREATE INDEX idx_llm_run_prompt_name ON LLM_RUN (Prompt_name);


Почему эта таблица нужна:
- удобно дебажить, почему бот задал вопрос или построил такой маршрут;
- можно показать жюри, что рекомендации строятся не случайно;
- если LLM вернула плохой JSON, ошибку можно сохранить и обработать.


6. TOP_USERS

Кэш топа пользователей.
Основной XP хранится в USER_PROFILE.Global_xp, но отдельная таблица нужна для недельных/месячных рейтингов, истории мест и быстрого вывода лидерборда в mini app.

CREATE TABLE TOP_USERS (
    Top_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Profile_id UUID NOT NULL REFERENCES USER_PROFILE(User_id) ON DELETE CASCADE,

    Period_type TEXT NOT NULL CHECK (
        Period_type IN ('all_time', 'monthly', 'weekly')
    ),                                      -- Общий, месячный или недельный топ
    Period_start DATE NOT NULL,             -- Начало периода. Для all_time: 1970-01-01
    Period_end DATE NOT NULL,               -- Конец периода. Для all_time: 9999-12-31

    Rank_place INT NOT NULL,                -- Место пользователя в топе
    Xp_value INT NOT NULL DEFAULT 0,         -- XP за выбранный период

    Streak_days INT NOT NULL DEFAULT 0,
    Completed_items INT NOT NULL DEFAULT 0,  -- Сколько материалов/заданий завершено за период
    Completed_roadmaps INT NOT NULL DEFAULT 0,

    Username_snapshot TEXT,                 -- Снапшот ника, чтобы топ не ломался при смене username
    Display_name_snapshot TEXT,

    Calculated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

    UNIQUE (Period_type, Period_start, Period_end, Profile_id),
    UNIQUE (Period_type, Period_start, Period_end, Rank_place)
);

CREATE INDEX idx_top_users_period_rank ON TOP_USERS (Period_type, Period_start, Period_end, Rank_place);
CREATE INDEX idx_top_users_profile ON TOP_USERS (Profile_id);
CREATE INDEX idx_top_users_xp ON TOP_USERS (Period_type, Xp_value DESC);


Почему эта таблица нужна:
- USER_PROFILE хранит текущий общий XP, но не хранит место пользователя в недельном/месячном сезоне;
- топ можно пересчитывать по расписанию и быстро показывать пользователю;
- можно сделать механику сезонов: "топ недели", "топ месяца", "топ за всё время";
- снапшоты username/display_name помогают сохранить корректный вид лидерборда, даже если пользователь поменяет профиль Telegram.


7. MOTIVATION_PUSH

Очередь мотивационных уведомлений.
Нужна для напоминаний о дедлайнах, streak, просроченных материалах и XP. Тон может быть "duolingo_aggressive", но пуши должны соблюдать лимиты и quiet hours.

CREATE TABLE MOTIVATION_PUSH (
    Push_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    Profile_id UUID NOT NULL REFERENCES USER_PROFILE(User_id) ON DELETE CASCADE,
    Roadmap_id UUID REFERENCES ROADMAP(Roadmap_id) ON DELETE CASCADE,
    Item_id UUID REFERENCES ROADMAP_ITEM(Item_id) ON DELETE SET NULL,

    Push_type TEXT NOT NULL CHECK (
        Push_type IN ('deadline_warning', 'deadline_expired', 'streak_risk', 'xp_opportunity', 'return_to_route', 'test_required')
    ),
    Tone TEXT NOT NULL DEFAULT 'duolingo_aggressive' CHECK (
        Tone IN ('soft', 'neutral', 'duolingo_aggressive')
    ),

    Message_text TEXT NOT NULL,
    Button_text TEXT,
    Button_payload JSONB NOT NULL DEFAULT '{}'::jsonb,

    Scheduled_at TIMESTAMPTZ NOT NULL,
    Sent_at TIMESTAMPTZ,
    Status TEXT NOT NULL DEFAULT 'planned' CHECK (
        Status IN ('planned', 'sent', 'cancelled', 'failed', 'skipped_by_quiet_hours', 'rate_limited')
    ),

    Created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_motivation_push_schedule ON MOTIVATION_PUSH (Status, Scheduled_at);
CREATE INDEX idx_motivation_push_profile ON MOTIVATION_PUSH (Profile_id);
CREATE INDEX idx_motivation_push_item ON MOTIVATION_PUSH (Item_id);


Почему эта таблица нужна:
- пуши должны планироваться, а не отправляться хаотично;
- можно ограничить частоту уведомлений и не нарушать quiet hours;
- можно делать агрессивный игровой тон, но оставлять контроль частоты и отключение;
- пуши помогают удерживать streak, дедлайны и прохождение мини-тестов.

Что НЕ добавляем в MVP:
- отдельный MATERIALS_CATALOG, потому что LLM с интернет-доступом сама ищет материалы;
- отдельные таблицы достижений, потому что на старте достаточно Global_xp, Streak_days, Verified_xp и Xp в ROADMAP_ITEM;
- отдельную таблицу сообщений диалога, если история пока хранится в FSM/Redis или в Profile_json.
